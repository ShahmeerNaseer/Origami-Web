💀 How to Use Skeleton Loaders

Skeleton loaders show a grey animated placeholder while content loads. Here's how to add them to your pages:

Step 1: Add Skeleton HTML (Before Content Loads)

Wrap your content area with a skeleton placeholder. When the real content loads, swap it out.

📍 Example: Map Skeleton Loader (contact.html)

<!-- SKELETON LOADER (shown while map is loading) -->
<div id="map-skeleton" class="skeleton skeleton-map"></div>

<!-- ACTUAL MAP (hidden initially) -->
<div id="map-content" class="content-loaded" style="display:none;">
    <iframe 
        src="https://www.google.com/maps/embed?pb=..."
        width="100%" height="100%" style="border:0;"
        allowfullscreen loading="lazy"
        onload="showMap()">
    </iframe>
</div>

<script>
function showMap() {
    document.getElementById('map-skeleton').style.display = 'none';
    const mapContent = document.getElementById('map-content');
    mapContent.style.display = 'block';
    // Small delay to trigger fade-in
    setTimeout(() => mapContent.classList.add('show'), 50);
}
</script>

🃏 Example: Card Skeleton Loader (categories.html or index.html)

<!-- SKELETON CARD -->
<div id="card-skeleton-1" class="skeleton-card">
    <div class="skeleton skeleton-image"></div>
    <div class="skeleton skeleton-title"></div>
    <div class="skeleton skeleton-text"></div>
    <div class="skeleton skeleton-text short"></div>
    <div class="skeleton skeleton-button"></div>
</div>

<!-- ACTUAL CARD (hidden initially) -->
<div id="card-content-1" class="content-loaded" style="display:none;">
    <div class="card">
        <img src="images/crane.png" alt="Paper Crane">
        <h4>Paper Crane</h4>
        <p>Learn to fold a beautiful paper crane.</p>
        <button>Learn More</button>
    </div>
</div>

<script>
// Simulate loading delay (remove this in production, use real data fetch)
setTimeout(() => {
    document.getElementById('card-skeleton-1').style.display = 'none';
    const content = document.getElementById('card-content-1');
    content.style.display = 'block';
    setTimeout(() => content.classList.add('show'), 50);
}, 1500); // 1.5 second fake loading time
</script>

🖼️ Example: Gallery Image Skeletons

<div class="col-md-4">

    <div id="gallery-content-1" class="content-loaded" style="display:none;">

<script>
function showGallery(id) {
    document.getElementById('gallery-skeleton-' + id).style.display = 'none';
    const content = document.getElementById('gallery-content-' + id);
    content.style.display = 'block';
    setTimeout(() => content.classList.add('show'), 50);
}
</script>

⚡ Pro Tips

  • Use onload on images/iframes to trigger the skeleton hide
  • Use setTimeout to simulate loading during development
  • For real data fetching, hide skeleton in the .then() or await after your fetch completes
  • The shimmer animation runs automatically — no JS needed for the animation itself!
  • Combine multiple skeleton elements to match your layout structure

🚀 Full Page Skeleton Example (for index.html)

<body>
    <!-- FULL PAGE SKELETON -->
    <div id="page-skeleton">
        <div class="skeleton" style="height:80px; width:100%; border-radius:0;"></div>
        <div class="skeleton" style="height:400px; width:100%; border-radius:0; margin:20px 0;"></div>
        <div class="container">
            <div class="row">
                <div class="col-md-4"><div class="skeleton-card"><div class="skeleton skeleton-image"></div><div class="skeleton skeleton-title"></div><div class="skeleton skeleton-text"></div></div></div>
                <div class="col-md-4"><div class="skeleton-card"><div class="skeleton skeleton-image"></div><div class="skeleton skeleton-title"></div><div class="skeleton skeleton-text"></div></div></div>
                <div class="col-md-4"><div class="skeleton-card"><div class="skeleton skeleton-image"></div><div class="skeleton skeleton-title"></div><div class="skeleton skeleton-text"></div></div></div>
            </div>
        </div>
    </div>

    <!-- ACTUAL PAGE CONTENT (hidden initially) -->
    <div id="page-content" style="display:none;">
        <!-- your actual page content here -->
    </div>

    <script>
    window.addEventListener('load', () => {
        setTimeout(() => {
            document.getElementById('page-skeleton').style.display = 'none';
            document.getElementById('page-content').style.display = 'block';
        }, 1000); // 1 second loading simulation
    });
    </script>
</body>
